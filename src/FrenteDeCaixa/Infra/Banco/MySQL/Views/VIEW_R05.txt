	CREATE
	VIEW pafecf.VIEW_R05
	AS
	(
		select 
		VC.ID AS VCID, 
		V.ID AS VID, 
		V.ITEM, 
		V.SERIE_ECF,
		V.GTIN, 
		V.CCF,
		V.COO,
		P.DESCRICAO_PDV, 
		V.QUANTIDADE, 
		V.CST, 
		V.PIS, 
		V.COFINS, 
		V.HASH_TRIPA, 
		V.HASH_INCREMENTO,
		U.NOME AS SIGLA_UNIDADE, 
		V.VALOR_UNITARIO, 
		V.DESCONTO, 
		V.ACRESCIMO, 
		V.CFOP, 
		V.TAXA_ICMS, 
		V.ID_ECF_PRODUTO, 
		V.TOTAL_ITEM, 
		V.TOTALIZADOR_PARCIAL, 
		V.CANCELADO, 
		P.IAT, 
		P.IPPT,
		P.ID_UNIDADE_PRODUTO AS ID_UNIDADE
	from 
		ECF_VENDA_CABECALHO VC, ECF_VENDA_DETALHE V, PRODUTO P, UNIDADE_PRODUTO U 
	where 
		VC.ID=V.ID_ECF_VENDA_CABECALHO
		and V.ID_ECF_PRODUTO=P.ID 
		and P.ID_UNIDADE_PRODUTO=U.ID 
	)
